import matplotlib.pyplot as plt
import numpy as np

f = np.pi

deltaxp = []

stepsize = 50000

time = np.linspace(0,1,10000)

def delta_xp(p,q,begin,end,step,finarr):
    for t in np.arange(begin,end,step):
        plus = p**2 + q**2
        minus = p**2 - q**2
        a1 = np.exp(-1 * plus * (1 - np.cos(2 * f * t)))
        a2 = np.exp(-1 * plus * (1 - np.cos(4 * f * t)))
        b1 = np.cos(plus * (np.sin(2 * f * t)))
        c1 = np.sin(plus * (np.sin(2 * f * t)))
        b2 = np.cos((2 * f * t) + (plus * np.sin(4 * f * t)))
        c2 = np.sin((2 * f * t) + (plus * np.sin(4 * f * t)))
    
        finx = a1 * ((p * b1) + (q * c1))
        finp = a1 * ((p * c1) - (q * b1))
    
        finx2 = 0.5 * (1 + plus + a2 * (((minus) * b2) + (2 * p * q) * c2))
        finp2 = 0.5 * (1 + plus - a2 * (((minus) * b2) + (2 * p * q) * c2))
    
        deltax = np.sqrt(finx2 - (finx**2))
        deltap = np.sqrt(finp2 - (finp**2))
    
        final = deltax * deltap
    
        finarr.append(final)
        
delta_xp(10,10,0,1,0.0001,deltaxp)  

plt.rcParams["figure.dpi"] = 200
        
plt.plot(time,deltaxp) 
plt.xlabel("$t = T_{rev}$")
plt.ylabel("$\Delta x \Delta p$")
plt.title("$p_1 , q_1 = 10$")
plt.show()

We know that the dip in the centre corresponds to the first fractional revival. We shall zoom into this area in the graph to understand it better.
